CREATE TABLE Produtos (
        id INT IDENTITY(1,1) PRIMARY KEY,
        nome NVARCHAR(255),
        descricao NVARCHAR(MAX),
        preco DECIMAL(18,2),
        categoria NVARCHAR(100),
        created_at DATETIME2 DEFAULT SYSDATETIME(),
        estoque INT NOT NULL DEFAULT 0,
        ativo BIT DEFAULT 1,
        image_url NVARCHAR(2083)
    )
